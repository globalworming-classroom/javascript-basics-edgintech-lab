Async Error Handling

When you work with async/await, things tend to look normal again 
const failingGet = () => {
    return new Promise((resolve, reject) => reject("failed"))
}

const doStuff = async () => {
    try {
        await failingGet()
    } catch (e) {
        console.error("caught" , e)
    } finally {
        console.log("cleanup")
    }
}
doStuff()
// -> caught failed
// -> cleanup
